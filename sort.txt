program sortowanie;
const
SIZE = 15;

type tab = array[1..SIZE] of integer;

procedure swap(var tablica : tab; i : integer; j: integer);
var
tmp : integer;
begin
tmp := tablica[i];
tablica[i] := tablica[j];
tablica[j] := tmp;
end;

function min(tablica : tab):integer;
var
tmp, i : integer;
begin
tmp := tablica[1];
for i:=1 to SIZE do
    begin
    if tmp > tablica[i] then
    tmp := tablica[i];
    end;
result:=i;   //zwraca numer tablicy o minimalnej wartosci
end;
function max(tablica : tab):integer;
var
tmp, i : integer;
begin
tmp := tablica[1];
for i:=1 to SIZE do
    begin
    if tmp < tablica[i] then
    tmp := tablica[i];
    end;
result:=tmp;
end;
